```python
import os
from logging.config import fileConfig
from alembic import context
from sqlalchemy import engine_from_config, pool

config = context.config

if config.config_file_name is not None:
    fileConfig(config.config_file_name)

# Override sqlalchemy.url from environment variable
database_url = os.environ.get('DATABASE_URL')
if database_url:
    config.set_main_option('sqlalchemy.url', database_url)

def run_migrations_online():
    connectable = engine_from_config(
        config.get_section(config.config_ini_section),
        prefix="sqlalchemy.",
        poolclass=pool.NullPool,
    )
    with connectable.connect() as connection:
        context.configure(connection=connection, target_metadata=None)
        with context.begin_transaction():
            context.run_migrations()

run_migrations_online()
```

Create `migrations/versions/001_initial_schema.py` per D-07 (manual Alembic migrations with raw SQL via op.execute).

The schema implements D-01 (normalized tables), D-02 (typed restaurant columns with JSONB for nested Google fields per Assumption A1), D-03 (nullable workspace_id), and D-04 (unique constraint for vote toggle):

```python
"""Initial schema: restaurants, polls, poll_options, votes

Revision ID: 001
Revises: None
Create Date: 2026-04-05
"""
from alembic import op

revision = '001'
down_revision = None
branch_labels = None
depends_on = None

def upgrade():
    op.execute("""
        CREATE TABLE restaurants (
            id SERIAL PRIMARY KEY,
            place_id VARCHAR(255) UNIQUE NOT NULL,
            name VARCHAR(255) NOT NULL,
            rating NUMERIC(2,1),
            price_level SMALLINT,
            url TEXT,
            website TEXT,
            emoji VARCHAR(64),
            geometry JSONB,
            photos JSONB,
            opening_hours JSONB,
            icon TEXT,
            vicinity TEXT,
            types TEXT[],
            user_ratings_total INTEGER,
            workspace_id VARCHAR(64),
            created_at TIMESTAMPTZ DEFAULT NOW(),
            updated_at TIMESTAMPTZ DEFAULT NOW()
        )
    """)
    op.execute("""
        CREATE TABLE polls (
            id SERIAL PRIMARY KEY,
            poll_date DATE NOT NULL,
            workspace_id VARCHAR(64),
            slack_channel_id VARCHAR(64),
            slack_message_ts VARCHAR(64),
            created_at TIMESTAMPTZ DEFAULT NOW(),
            UNIQUE(poll_date, workspace_id)
        )
    """)
    op.execute("""
        CREATE TABLE poll_options (
            id SERIAL PRIMARY KEY,
            poll_id INTEGER NOT NULL REFERENCES polls(id) ON DELETE CASCADE,
            restaurant_id INTEGER NOT NULL REFERENCES restaurants(id),
            display_order SMALLINT DEFAULT 0,
            UNIQUE(poll_id, restaurant_id)
        )
    """)
    op.execute("""
        CREATE TABLE votes (
            id SERIAL PRIMARY KEY,
            poll_option_id INTEGER NOT NULL REFERENCES poll_options(id) ON DELETE CASCADE,
            user_id VARCHAR(64) NOT NULL,
            created_at TIMESTAMPTZ DEFAULT NOW(),
            UNIQUE(poll_option_id, user_id)
        )
    """)

def downgrade():
    op.execute("DROP TABLE IF EXISTS votes CASCADE")
    op.execute("DROP TABLE IF EXISTS poll_options CASCADE")
    op.execute("DROP TABLE IF EXISTS polls CASCADE")
    op.execute("DROP TABLE IF EXISTS restaurants CASCADE")
```

Key design notes:
- restaurants.place_id is UNIQUE NOT NULL (Google Places identifier, used for upsert)
- restaurants.geometry, photos, opening_hours are JSONB (nested Google API structures per Pitfall 6 / Assumption A1)
- All other restaurant fields are typed columns per D-02
- workspace_id is nullable VARCHAR(64) per D-03 (Phase 2 adds RLS)
- polls has UNIQUE(poll_date, workspace_id) for one poll per day per workspace
- votes has UNIQUE(poll_option_id, user_id) per D-04 preventing duplicate votes
- Foreign keys with ON DELETE CASCADE for poll_options and votes
